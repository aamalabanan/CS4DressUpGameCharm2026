public class RatingSystem {
    private int starRating;

    public int determineStarRating(int score) {
        if (score >= 81) {
            return 5;
        } else if (score >= 61) {
            return 4;
        } else if (score >= 41) {
            return 3;
        } else if (score >= 21) {
            return 2;
        } else {
            return 1;
        }
    }

    public int getStarRating() {
        return starRating;
    }

    public void setStarRating(int starRating) {
        this.starRating = starRating;
    }
}
