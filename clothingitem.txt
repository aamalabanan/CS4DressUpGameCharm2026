public class ClothingItem {
    private String type;
    private String style;
    private Theme theme;
    private boolean owned;

    public ClothingItem(String type, String style, Theme theme) {
        this.type = type;
        this.style = style;
        this.theme = theme;
        this.owned = false;
    }

    public String getType() {
        return type;
    }

    public String getStyle() {
        return style;
    }

    public Theme getTheme() {
        return theme;
    }

    public boolean isOwned() {
        return owned;
    }

    public void setOwned(boolean owned) {
        this.owned = owned;
    }
}
