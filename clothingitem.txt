public class ClothingItem{
    private String style;
    private String type;
    private int price;
    private boolean owned;

    public ClothingItem(String style, String type, int price) {
        this.style = style;
        this.type = type;
        this.price = price;
        this.owned = false;
    }

    public int getPrice() {
        return price;
    }

    public boolean isOwned() {
        return owned;
    }

    public void setOwned(boolean owned) {
        this.owned = owned;
    }
